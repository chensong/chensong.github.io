## 1. 启动参数

```sh
traefik --api --log.level=DEBUG \
      # Entrypoints
      --entrypoints.web.address=:80 \
      --entrypoints.web.http.redirections.entrypoint.to=websecure \
      --entrypoints.web.http.redirections.entrypoint.scheme=https \
      --entrypoints.websecure.address=:443 \
      # dnschallenge
      --entrypoints.websecure.http.tls.certresolver=mydnschallenge \
      --entrypoints.websecure.http.tls.domains[0].main=local.example.com \
      --entrypoints.websecure.http.tls.domains[0].sans=*.local.example.com \
      --entrypoints.other.address=:9090 \
      # acme
      --certificatesresolvers.mydnschallenge.acme.email=mymail@gmail.com \
      --certificatesresolvers.mydnschallenge.acme.dnschallenge.provider=dnspod \
      --certificatesresolvers.mydnschallenge.acme.storage=acme.json \
      # providers.file
      --providers.file.filename=traefik_conf.yaml
```

```sh
traefik --api --config traefik.yaml
```
```yaml
# traefik.yaml
entryPoints:
  web:
    address: ":80"
    http:
      redirections:
        entryPoint:
          to: websecure
          scheme: https
  websecure:
    address: ":443"
    http:
      tls:
        certResolver: myresolver
        domains:
          - main: "local.example.com"
            sans: 
              - "*.local.example.com"
  other:
    address: ":9090"

api:
  insecure: true

providers:
  # Enable the file provider to define routers / middlewares / services in file
  file:
    filename: traefik_conf.yaml
    watch: true

certificatesResolvers:
  mydnschallenge:
    # Enable ACME (Let's Encrypt): automatic SSL.
    acme:

      # Email address used for registration.
      #
      # Required
      #
      email: "mymail@gmail.com"

      # File or key used for certificates storage.
      #
      # Required
      #
      storage: acme.json

      # CA server to use.
      # Uncomment the line to use Let's Encrypt's staging server,
      # leave commented to go to prod.
      #
      # Optional
      # Default: "https://acme-v02.api.letsencrypt.org/directory"
      #
      # caServer: "https://acme-staging-v02.api.letsencrypt.org/directory"

      # Preferred chain to use.
      #
      # If the CA offers multiple certificate chains, prefer the chain with an issuer matching this Subject Common Name.
      # If no match, the default offered chain will be used.
      #
      # Optional
      # Default: ""
      #
      # preferredChain: 'ISRG Root X1'

      # KeyType to use.
      #
      # Optional
      # Default: "RSA4096"
      #
      # Available values : "EC256", "EC384", "RSA2048", "RSA4096", "RSA8192"
      #
      # keyType: RSA4096

      # Use a TLS-ALPN-01 ACME challenge.
      #
      # Optional (but recommended)
      #
      # tlsChallenge:

      # Use a HTTP-01 ACME challenge.
      #
      # Optional
      #
      # httpChallenge:

        # EntryPoint to use for the HTTP-01 challenges.
        #
        # Required
        #
        # entryPoint: web

      # Use a DNS-01 ACME challenge rather than HTTP-01 challenge.
      # Note: mandatory for wildcard certificate generation.
      #
      # Optional
      #
      # dnsChallenge:

        # DNS provider used.
        #
        # Required
        #
        # provider: digitalocean

        # By default, the provider will verify the TXT DNS challenge record before letting ACME verify.
        # If delayBeforeCheck is greater than zero, this check is delayed for the configured duration in seconds.
        # Useful if internal networks block external DNS queries.
        #
        # Optional
        # Default: 0
        #
        # delayBeforeCheck: 0

        # Use following DNS servers to resolve the FQDN authority.
        #
        # Optional
        # Default: empty
        #
        # resolvers
        # - "1.1.1.1:53"
        # - "8.8.8.8:53"

        # Disable the DNS propagation checks before notifying ACME that the DNS challenge is ready.
        #

        # Increase the risk of reaching Let's Encrypt's rate limits.
        #
        # Optional
        # Default: false
        #
        # disablePropagationCheck: true    
      dnsChallenge:
          # 更多 DNS 厂商配置 https://doc.traefik.io/traefik/https/acme/#dnschallenge
          # 使用 dnspod 需要配置环境变量: 
          # DNSPOD_API_KEY=token_id,token_val
          # DNSPOD_HTTP_TIMEOUT=30
          provider: dnspod
          delayBeforeCheck: 0
```

## 2. 路由配置

```yaml
http:
  routers:
    dockerHub:
      rule: Host(`hub.local.example.com`)
      service: dockerHub
    dockerRegistry:
      rule: Host(`registry.local.example.com`)
      service: dockerRegistry
    # Traefik Dashboard
    dashboard:
      rule: Host(`dashboard.local.example.com`)
      service: dashboard@internal
    dashboard-api:
      rule: Host(`dashboard.local.example.com`) && (PathPrefix(`/api`) || PathPrefix(`/dashboard`))
      service: api@internal      
      middlewares:
        - auth
  middlewares:
    auth:
      basicAuth:
        # 用 htpasswd 生成密码
        users:
          - "test:$apr1$WBN94bmo$hal.uM5FWDzecx3LNnpcF/"
  services:
    dockerHub:
      loadBalancer:
        servers:
          - url: http://localhost:8000
    dockerRegistry:
      loadBalancer:
        servers:
          - url: http://localhost:5000
```
